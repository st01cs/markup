```typescript
/**
 * Slugifies heading text for use as an ID attribute.
 * Handles duplicates by appending -1, -2, etc.
 * D-03: text.toLowerCase().replace(/[^\w]+/g, '-'), duplicate detection with numeric suffix
 */
export function slugify(text: string, existingIds: string[] = []): string {
  let slug = text.toLowerCase().replace(/[^\w]+/g, '-').replace(/^-|-$/g, '');
  if (!slug) {
    slug = 'heading';
  }

  let uniqueId = slug;
  let counter = 1;
  while (existingIds.includes(uniqueId)) {
    uniqueId = `${slug}-${counter++}`;
  }
  return uniqueId;
}
```

```typescript
// Track headings during token walk for TOC extraction
const headings: Heading[] = [];

marked.use({
  walkTokens(token) {
    if (token.type === 'heading' && token.depth <= 3) {
      const t = token as Tokens.Heading;
      const id = slugify(t.text, headings.map(h => h.id));
      headings.push({ level: t.depth as 1 | 2 | 3, text: t.text, id });
      // Store id on token for renderer to use
      (token as any).headingId = id;
    }
  },
  renderer: {
    heading(token) {
      const id = (token as any).headingId || slugify(token.text);
      const text = renderInlineTokens(token.tokens);
      return `<h${token.depth} id="${id}">${text}</h${token.depth}>`;
    }
  }
});
```

Note: The `renderInlineTokens` function does not currently exist. You need to create it to render inline tokens (text, bold, italic, etc.) from `token.tokens`. Look at how marked handles inline token rendering - typically it uses `marked.parseInline()` or similar. For a simple implementation, if the heading has no inline tokens (just plain text), use `token.text` directly.

For headings with inline tokens (e.g., "**bold** heading"), you need to render them properly. The standard marked approach is to use `this.parser.parseInline(token.tokens)` but since we're using the renderer directly, you may need a simpler approach:

```typescript
function renderInlineTokens(tokens: Token[]): string {
  if (!tokens || tokens.length === 0) return '';
  // For simple text tokens, return the text directly
  // For complex inline tokens, render appropriately
  return tokens.map(token => {
    if (token.type === 'text') {
      return token.text;
    }
    // Handle other inline types as needed
    return token.raw;
  }).join('');
}
```

This is a simplified approach. If headings have complex inline formatting (bold, italic, code), the `renderInlineTokens` function should handle them properly. Test with headings like "## Hello **World**" to verify inline formatting is preserved.

```typescript
export function renderMarkdown(raw: string): RenderResult {
  const errors: string[] = [];
  headings.length = 0; // Reset headings array before each parse (D-08)

  if (!raw || raw.trim() === '') {
    return { html: '', errors: [], headings: [] };

    ALLOWED_TAGS,
    ALLOWED_ATTR,
    ALLOW_DATA_ATTR: false,
  });

  return { html: clean, errors, headings }; // Return headings array (TOC-01)
}
```
