3. Add TOC panel open/close/render functions BEFORE the event listeners section:

```typescript
function openTocPanel() {
  isTocPanelOpen = true;
  tocPanel.classList.add('open');
  tocPanel.style.display = 'flex';
  fabWidget.classList.remove('expanded');
  // Focus first heading item
  const firstItem = tocList.querySelector('.toc-item') as HTMLButtonElement | null;
  if (firstItem) {
    firstItem.focus();
  }
}

function closeTocPanel() {
  isTocPanelOpen = false;
  tocPanel.classList.remove('open');
  // Return focus to FAB toggle per D-24
  fabToc.focus();
}

function renderTocList(headings: Heading[]) {
  tocList.innerHTML = '';
  for (const heading of headings) {
    const button = document.createElement('button');
    button.className = 'toc-item';
    button.setAttribute('data-level', String(heading.level));
    button.textContent = heading.text;
    button.addEventListener('click', () => {
      const target = document.getElementById(heading.id);
      if (target) {
        closeTocPanel();
        target.scrollIntoView({ behavior: 'smooth', block: 'start' });
      }
    });
    tocList.appendChild(button);
  }
}
```
